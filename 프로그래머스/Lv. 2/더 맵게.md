## O(n^2logn)
```Python3
def solution(scoville, K):
    #섞은 스코빌 지수=가장낮+(두번째로낮*2)
    #리스트 하나 남을 때 까지
    cnt=0
    while True:
        if all(x>=K for x in scoville):
            return cnt
        if len(scoville)==1:
            return -1
        
        cnt+=1
        scoville.sort(reverse=True) 
        a=scoville[-1]+scoville[-2]*2
        scoville.append(a)
        scoville.sort(reverse=True) 
        del scoville[-2:]
```

## O(nlogn)
```Python3
import heapq
def solution(scoville,K):
    heapq.heapify(scoville) #최소 힙
    cnt=0
    
    while len(scoville)>1:
        if scoville[0]>=K:
            return cnt
        a=heapq.heappop(scoville)
        b=heapq.heappop(scoville)
    
        new=a+b*2
        heapq.heappush(scoville,new)
        cnt+=1
    if scoville[0]>=K:
        return cnt
    else:
        return -1
```
