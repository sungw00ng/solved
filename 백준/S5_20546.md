`   기적의 매매법   ` <br>
https://www.acmicpc.net/problem/20546 <br>
```python
import sys
input=sys.stdin.readline

현금=int(input())
주가=list(map(int, input().strip().split()))

준현_현금=현금
성민_현금=현금
준현_보유주식수=0
성민_보유주식수=0

for i in range(14):  
    # 준현의 BNP 매매법
    준현_보유주식수+=준현_현금//주가[i]
    준현_현금%=주가[i]

    #성민의 33매매법
    if i>=3:
        # 전량 매수
        if 주가[i-3]>주가[i-2]>주가[i-1]:
            성민_보유주식수+=성민_현금//주가[i]
            성민_현금%=주가[i]

        # 전량 매도
        if 주가[i-3]<주가[i-2] < 주가[i-1]:
            성민_현금+=성민_보유주식수*주가[i]
            성민_보유주식수= 0

# 1월 14일 자산 계산
준현_자산=준현_현금+준현_보유주식수*주가[13]
성민_자산=성민_현금+성민_보유주식수*주가[13]

if 준현_자산>성민_자산:
    print("BNP")
elif 준현_자산<성민_자산:
    print("TIMING")
else:
    print("SAMESAME")

```
